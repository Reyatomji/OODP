using namespace std;
class Circle
{   float radius;
    public:
        void getdata()
        {cin>>radius;}

        void perimeter()
        {      
            cout<<"Perimeter :"<<2*3.14*radius<<endl;
        }

        void area()
        {
            cout<<"Area :"<<3.14*radius*radius<<endl;
        }

};

int main()
{  
    Circle c1;
    c1.getdata();
    c1.area();
    c1.perimeter();
    return 0;
}
